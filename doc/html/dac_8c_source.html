36 void dac_init() {
37  // This assumes CS for DAC is RB4 (pin 11)
38  // set CS high initially
39  mPORTBSetPinsDigitalOut(BIT_4);
40  DAC_SET_CS();
41 
42  // SCK2 is pin 26. SDO2 must be assigned using PPS.
43  // The following PPS mapping is already done by ioe_init, so it's redundant
44  // (but harmless) if the I/O Expander has been initialized.
45  PPSOutput(2, RPB5, SDO2); // RPB5 (pin 14) --> SDO2 (MOSI)
46 
47  // The following SpiChnOpen is redundant if using I/O expander. ioe_init()
48  // does the same thing, only using MODE8, but that's fine, as when talking
49  // to the DAC we always set MODE16 first.
50  SpiChnOpen(SPI_CHANNEL2, SPI_OPEN_ON | SPI_OPEN_MODE16 |
51  SPI_OPEN_MSTEN | SPI_OPEN_CKE_REV ,
52  PBCLK / 10000000); // We use an SPI clock divisor that gives
53  // 10 MHz because the I/O expander can't
54  // handle anything faster.
55 }

68 inline void dac_write(uint16_t msg) {
69  // wait until ready
70  while (!SpiChnTxBuffEmpty(DAC_SPI_CHN)) { ; }
71  SPI_Mode16();
72  DAC_CLEAR_CS(); // CS low to start transaction
73  SpiChnWriteC(DAC_SPI_CHN, msg);
74  while (SpiChnIsBusy(DAC_SPI_CHN)) { ; } // wait for end of transaction
75  DAC_SET_CS(); // CS high to end transaction
76  // need to read SPI channel to avoid possibly confusing I/O expander
77  SpiChnReadC(DAC_SPI_CHN); // (ignore return value)
78 }
