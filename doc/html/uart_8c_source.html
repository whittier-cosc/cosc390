29 void uart_init() {
30  CFGCONbits.IOLOCK = 0;
31  U1RXR = 0; // Map RPA2 (pin 9) to U1RX
32  RPB3R = 1; // Map RPB3 (pin 7) to U1TX
33  CFGCONbits.IOLOCK = 1;
34 
35  // Set baud to BAUDRATE
36  U1MODEbits.BRGH = 0; // High-speed mode disabled
37  // With PBCLK = SYSCLK = 40 M, we have U1BRG = 259, giving
38  // baud rate = 9615.4 (see DS61107F, Table 21-2).
39  U1BRG = (PBCLK / BAUDRATE) / 16 - 1;
40  // 8 bit, no parity bit, 1 stop bit (8N1)
41  U1MODEbits.PDSEL = 0;
42  U1MODEbits.STSEL = 0;
43 
44  // Enable TX & RX, taking over U1RX/TX pins
45  U1STAbits.UTXEN = 1;
46  U1STAbits.URXEN = 1;
47  // Do not enable RTS or CTS
48  U1MODEbits.UEN = 0;
49 
50  // Enable the UART
51  U1MODEbits.ON = 1;
52 }

66 void uart_read(char *message, int maxLength) {
67  char data = 0;
68  int complete = 0, num_bytes = 0;
69  // loop until you get a '\r' or '\n'
70  while (!complete) {
71  if (U1STAbits.URXDA) { // if data is available
72  data = U1RXREG; // read the data
73  if ((data == '\r') || (data == '\n')) {
74  complete = 1;
75  } else {
76  message[num_bytes] = data;
77  ++num_bytes;
78  // overwrite from beginning of array if we would overflow
79  if (num_bytes >= maxLength) {
80  num_bytes = 0;
81  }
82  }
83  }
84  }
85  // end the string
86  message[num_bytes] = '\0';
87 }
