59 void ioe_init() {
60  mPORTBSetPinsDigitalOut(BIT_9); // use RPB9 (pin 18) as CS
61  IOE_SET_CS(); // CS high initially
62 
63  // These pin mappings assume that IOE_SPI_CHN is SPI_CHANNEL2!
64  PPSOutput(2, RPB5, SDO2); // use RPB5 (pin 14) for SDO2 (MOSI)
65  PPSInput(3, SDI2, RPA4); // use RPA4 (pin 12) for SDI2 (MISO)

68  SPI_OPEN_ON | SPI_OPEN_MODE8 | SPI_OPEN_MSTEN | SPI_OPEN_CKE_REV,

195 inline void ioe_write(unsigned char reg_addr, unsigned char data) {
196 
197  // wait until ready
198  while (!SpiChnTxBuffEmpty(IOE_SPI_CHN)) { ; }
199  SPI_Mode8();
200 
201  // CS low to start transaction
202  IOE_CLEAR_CS();
203 
204  // opcode and hw address (Should always be 0b0100000)
205  SpiChnWriteC(IOE_SPI_CHN, IOE_OPCODE_HEADER | IOE_WRITE);
206  while (SpiChnIsBusy(IOE_SPI_CHN)) { ; } // wait for byte to be sent
207  SpiChnReadC(IOE_SPI_CHN); // ignore junk return value
208 
209  // register address
210  SpiChnWriteC(IOE_SPI_CHN, reg_addr);
211  while (SpiChnIsBusy(IOE_SPI_CHN)) { ; }
212  SpiChnReadC(IOE_SPI_CHN);
213 
214  // one byte of data
215  SpiChnWriteC(IOE_SPI_CHN, data);
216  while (SpiChnIsBusy(IOE_SPI_CHN)) { ; }
217  SpiChnReadC(IOE_SPI_CHN);
218 
219  // CS high to end transaction
220  IOE_SET_CS();
221 
222 }

231 inline unsigned char ioe_read(unsigned char reg_addr) {
232  unsigned char data;
233 
234  // wait until ready
235  while (!SpiChnTxBuffEmpty(IOE_SPI_CHN)) { ; }
236  SPI_Mode8();
237 
238  // CS low to start transaction
239  IOE_CLEAR_CS();
240 
241  // opcode and hw address (Should always be 0b0100000)
242  SpiChnWriteC(IOE_SPI_CHN, IOE_OPCODE_HEADER | IOE_READ);
243  while (SpiChnIsBusy(IOE_SPI_CHN)) { ; } // wait for byte to be sent
244  data = SpiChnReadC(IOE_SPI_CHN); // junk
245 
246  // register address
247  SpiChnWriteC(IOE_SPI_CHN, reg_addr);
248  while (SpiChnIsBusy(IOE_SPI_CHN)) { ; }
249  data = SpiChnReadC(IOE_SPI_CHN); // junk
250 
251  // tx one byte of dummy data so we can rx real data
252  SpiChnWriteC(IOE_SPI_CHN, data);
253  while (SpiChnIsBusy(IOE_SPI_CHN)) { ; }
254  data = SpiChnReadC(IOE_SPI_CHN); // the byte we want
255 
256  // CS high to end transaction
257  IOE_SET_CS();
258 
259  return data;
260 }
