127 int8_t amp_getGain() {
128  int8_t gain = (int8_t) read8(TPA2016_GAIN);
129  gain = gain << 2;
130  if ((gain & 0x80) > 0)
131  gain = (gain >> 2) | 0xC0; // it's a negative value
132  else
133  gain = gain >> 2; // it's a positive value
134  return gain;

253 static uint8_t read8(uint8_t address) {
254  bool success;
255  uint8_t data;
256  I2C_7_BIT_ADDRESS SlaveAddress;
257  I2C_FORMAT_7_BIT_ADDRESS(SlaveAddress, TPA2016_I2CADDR, I2C_WRITE);
258 
259 // debug_log("read8: start_transfer\n");
260 
261  // Start the transfer to write data to the TPA2016
262  if(!start_transfer(false)) {
263  debug_log("read8: start_transfer failed\n");
264  while(1) { ; }
265  }
266 // debug_log("read8: transfer byte 1\n");
267  // Send slave address and register address
268  success = transmit_byte(SlaveAddress.byte);
269  if (!success) {
270  debug_log("read8: transmitting slave address byte failed\n");
271  while(1) { ; }
272  }
273 // debug_log("read8: transfer byte 2\n");
274  success = transmit_byte(address);
275  if (!success) {
276  debug_log("read8: transmitting slave address byte failed\n");
277  while(1) { ; }
278  }
279 // debug_log("read8: restart\n");
280  // Restart to do the read
281  if(!start_transfer(true)) {
282  debug_log("read8: start_transfer (restart) failed\n");
283  while(1) { ; }
284  }
285 // debug_log("read8: transmit slave addr (switch to read)\n");
286  // Transmit the slave address with the READ bit set
287  I2C_FORMAT_7_BIT_ADDRESS(SlaveAddress, TPA2016_I2CADDR, I2C_READ);
288  success = transmit_byte(SlaveAddress.byte);
289  if (!success) {
290  debug_log("read8: transmitting slave address byte (switch to read) failed\n");
291  while(1) { ; }
292  }
293 // debug_log("read8: enable receiver\n");
294  if (I2CReceiverEnable(TPA2016_I2C_BUS, true) == I2C_RECEIVE_OVERFLOW) {
295  debug_log("read8: receiver enable failed (receive overflow)\n");
296  while(1) { ; }
297  }
298 // debug_log("read8: wait for recv'd data available\n");
299  while(!I2CReceivedDataIsAvailable(TPA2016_I2C_BUS)) { ; }
300 // debug_log("read8: send nack\n");
301  I2CAcknowledgeByte(TPA2016_I2C_BUS, false); // send NACK after receiving final (and only) byte!
302 // debug_log("read8: get byte\n");
303  data = I2CGetByte(TPA2016_I2C_BUS);
304 
305  delay(); // kludge -- the following stop_transfer() hangs otherwise
306  stop_transfer();
307 
308  return data;
309 }

349 /*
350  * @brief Starts (or restarts) a transfer to/from the TPA2016
351  *
352  * This routine starts (or restarts) a transfer to/from the TPA2016,
353  * waiting (in a blocking loop) until the start (or re-start) condition
354  * has completed.
355  *
356  * This is a blocking routine that waits for the bus to be idle and the
357  * Start (or Restart) signal to complete.
358  *
359  * @pre The I2C module must have been initialized.
360  * @param restart
361  * if false, send a "Start" condition
362  if true, send a "Restart" condition
363  * @return true if successful, false if a collision occured during Start
364  * signaling
365  */
366 static bool start_transfer( bool restart ) {
367  I2C_STATUS status;
368 
369  // Send the Start (or Restart) signal
370  if(restart) {
371  I2CRepeatStart(TPA2016_I2C_BUS);
372  }
373  else {
374  // Wait for the bus to be idle, then start the transfer
375 // debug_log("start_transfer: wait for bus idle\n");
376  while(!I2CBusIsIdle(TPA2016_I2C_BUS)) { ; }
377 
378  if(I2CStart(TPA2016_I2C_BUS) != I2C_SUCCESS) {
379  debug_log("Error: Bus collision during transfer Start\n");
380  return false;
381  }
382  }
383 
384  // Wait for the signal to complete
385 // debug_log("start_transfer: wait for status == START\n");
386  do {
387  status = I2CGetStatus(TPA2016_I2C_BUS);
388  } while (!(status & I2C_START));
389 
390  return true;
391 }

411  * \brief Transmit a byte via I2C to the TPA2016
412  * \param data
413  * the byte to be transmitted
414  * \return true if successful
415  * false if unsuccessful
416  */
417 static bool transmit_byte(uint8_t data) {
418  // Wait for the transmitter to be ready
419 // debug_log("transmit_byte: wait for transmitter ready\n");
420  while(!I2CTransmitterIsReady(TPA2016_I2C_BUS)) { ; }
421 
422  // Transmit the byte
423 // debug_log("transmit_byte: send byte\n");
424  if(I2CSendByte(TPA2016_I2C_BUS, data) == I2C_MASTER_BUS_COLLISION) {
425  debug_log("Error: I2C Master Bus Collision\n");
426  return false;
427  }
428 
429  // Wait for the transmission to finish
430 // debug_log("transmit_byte: wait for transmission to finish\n");
431  while(!I2CTransmissionHasCompleted(TPA2016_I2C_BUS)) { ; }
432 
433 // debug_log("check if byte ack'd\n");
434  if(!I2CByteWasAcknowledged(TPA2016_I2C_BUS)) {
435  debug_log("Error: Sent byte was not acknowledged\n");
436  return false;
437  }
438  return true;
439 }
