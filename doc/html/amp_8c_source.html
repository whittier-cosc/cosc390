364 bool start_transfer( bool restart ) {
365  I2C_STATUS status;
366 
367  // Send the Start (or Restart) signal
368  if(restart) {
369  I2CRepeatStart(TPA2016_I2C_BUS);
370  }
371  else {
372  // Wait for the bus to be idle, then start the transfer
373 // debug_log("start_transfer: wait for bus idle\n");
374  while(!I2CBusIsIdle(TPA2016_I2C_BUS)) { ; }
375 
376  if(I2CStart(TPA2016_I2C_BUS) != I2C_SUCCESS) {
377  debug_log("Error: Bus collision during transfer Start\n");
378  return false;
379  }
380  }

386  } while (!(status & I2C_START));

418  while(!I2CTransmitterIsReady(TPA2016_I2C_BUS)) { ; }
