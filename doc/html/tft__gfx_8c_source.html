182 void tft_drawLine(short x0, short y0,
183  short x1, short y1,
184  unsigned short color) {
185 
186  // Bresenham's algorithm - thx wikpedia
187  short steep = abs(y1 - y0) > abs(x1 - x0);
188  if (steep) {
189  swap(x0, y0);
190  swap(x1, y1);
191  }
192 
193  if (x0 > x1) {
194  swap(x0, x1);
195  swap(y0, y1);
196  }
197 
198  short dx, dy;
199  dx = x1 - x0;
200  dy = abs(y1 - y0);
201 
202  short err = dx / 2;
203  short ystep;
204 
205  if (y0 < y1) {
206  ystep = 1;
207  } else {
208  ystep = -1;
209  }
210 
211  for (; x0 <= x1; x0++) {
212  if (steep) {
213  tft_drawPixel(y0, x0, color);
214  } else {
215  tft_drawPixel(x0, y0, color);
216  }
217  err -= dy;
218  if (err < 0) {
219  y0 += ystep;
220  err += dx;
221  }
222  }
223 }

284 void tft_fillTriangle (short x0, short y0,
285  short x1, short y1,
286  short x2, short y2,
287  unsigned short color) {
288  short a, b, y, last;
289 
290  // Sort coordinates by Y order (y2 >= y1 >= y0)
291  if (y0 > y1) {
292  swap(y0, y1); swap(x0, x1);
293  }
294  if (y1 > y2) {
295  swap(y2, y1); swap(x2, x1);
296  }
297  if (y0 > y1) {
298  swap(y0, y1); swap(x0, x1);
299  }
300 
301  if(y0 == y2) { // Handle awkward all-on-same-line case as its own thing
302  a = b = x0;
303  if(x1 < a) a = x1;
304  else if(x1 > b) b = x1;
305  if(x2 < a) a = x2;
306  else if(x2 > b) b = x2;
307  tft_drawFastHLine(a, y0, b-a+1, color);
308  return;
309  }
310 
311  short
312  dx01 = x1 - x0,
313  dy01 = y1 - y0,
314  dx02 = x2 - x0,
315  dy02 = y2 - y0,
316  dx12 = x2 - x1,
317  dy12 = y2 - y1,
318  sa = 0,
319  sb = 0;
320 
321  // For upper part of triangle, find scanline crossings for segments
322  // 0-1 and 0-2. If y1=y2 (flat-bottomed triangle), the scanline y1
323  // is included here (and second loop will be skipped, avoiding a /0
324  // error there), otherwise scanline y1 is skipped here and handled
325  // in the second loop...which also avoids a /0 error here if y0=y1
326  // (flat-topped triangle).
327  if (y1 == y2)
328  last = y1; // Include y1 scanline
329  else
330  last = y1-1; // Skip it
331 
332  for (y=y0; y<=last; y++) {
333  a = x0 + sa / dy01;
334  b = x0 + sb / dy02;
335  sa += dx01;
336  sb += dx02;
337  /* longhand:
338  a = x0 + (x1 - x0) * (y - y0) / (y1 - y0);
339  b = x0 + (x2 - x0) * (y - y0) / (y2 - y0);
340  */
341  if (a > b)
342  swap(a,b);
343  tft_drawFastHLine(a, y, b-a+1, color);
344  }
345 
346  // For lower part of triangle, find scanline crossings for segments
347  // 0-2 and 1-2. This loop is skipped if y1=y2.
348  sa = dx12 * (y - y1);
349  sb = dx02 * (y - y0);
350  for (; y<=y2; y++) {
351  a = x1 + sa / dy12;
352  b = x0 + sb / dy02;
353  sa += dx12;
354  sb += dx02;
355  /* longhand:
356  a = x1 + (x2 - x1) * (y - y1) / (y2 - y1);
357  b = x0 + (x2 - x0) * (y - y0) / (y2 - y0);
358  */
359  if(a > b) swap(a,b);
360  tft_drawFastHLine(a, y, b-a+1, color);
361  }
362 }
